\newglossaryentry{urn}{
    name={URN},
    description={do inglês \emph{Uniform Resource Name}; Nome Uniforme
    de Recursos, é o nome histórico dado a uma \glslink{uri}{\glsentryname{uri}}
    que usa o esquema ``urn:''. Sua sintaxe é ``urn:<NID>:<NSS>'', onde ``urn:'' é

    ``<NSS>'', que é a \glslink{string}{\glsentryname{string}} específica do espaço de
    nomes usado},
    first={nome uniforme de recursos (\glsentryname{urn})},
    firstplural={nomes uniformes de recursos (\glsentryname{urn})},
    long={nome uniforme de recursos},
    plural={\glsentrylong{urn}s}
